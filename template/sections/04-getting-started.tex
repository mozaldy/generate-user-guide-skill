\subsection{Signing in}

\begin{ugTask}{Signing in}
  \begin{ugSteps}
    \item Open APP\_NAME in your web browser.
    \ugScreenshot[0.55]{screenshots/01-open-login.png}{The login page or landing page}
    \item Click \ugButton{Sign In} or \ugButton{Login with NIK}.
    \ugScreenshot[0.35]{screenshots/02-login-button.png}{The sign-in button}
    \item Type your email address into the \ugField{Email} or \ugField{Username or NIK} field.
    \ugScreenshot[0.55]{screenshots/03-username-field.png}{The username or NIK field}
    \item Type your password into the \ugField{Password} field.
    \ugScreenshot[0.55]{screenshots/04-password-field.png}{The password field}
    \item Click \ugButton{Sign In}.
    \ugScreenshot[0.35]{screenshots/05-signin-button.png}{The final sign-in button}
  \end{ugSteps}
\end{ugTask}
