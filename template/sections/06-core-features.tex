\subsection{Managing FEATURE\_AREA}

\begin{ugModule}{FEATURE\_AREA}
  Short description of this area.

  \textbf{What you can do:}
  \begin{itemize}
    \item \textbf{Add an item} --- click \ugButton{Add Item}.
    \item \textbf{Edit an item} --- click the three-dot button (\textbf{\ldots})
      at the end of any row and choose \textit{Edit}.
    \item \textbf{Delete an item} --- click the three-dot button and choose
      \textit{Delete}. You will be asked to confirm.
  \end{itemize}

  When adding or editing, fill in these fields:

  \begin{center}
  \begin{tabular}{p{3cm}p{4cm}p{6cm}}
    \toprule
    \textbf{Field} & \textbf{Required?} & \textbf{What to enter} \\
    \midrule
    \ugField{Title}       & Yes & A short name for this item \\
    \ugField{Description} & Yes & A short summary \\
    \ugField{Order}       & Yes & A whole number for position in the list \\
    \bottomrule
  \end{tabular}
  \end{center}
\end{ugModule}
