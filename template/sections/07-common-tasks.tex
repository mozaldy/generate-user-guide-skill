\section{Common Tasks}

% 3--5 most common tasks a user will do. Each task is a ugTask{} block with
% ugSteps inside. Use \ugButton, \ugField, \ugMenu inline to reference UI
% elements --- if a matching PNG was captured, the inline screenshot renders
% automatically.

\begin{ugTask}{Add a new item}
  \begin{ugSteps}
    \item Click \ugMenu{Section} in the left-hand menu.
    \item Click \ugButton{Add Item}.
    \item Enter a \ugField{Title}, a \ugField{Description}, and an
      \ugField{Order} number.
    \item Click \ugButton{Save}.
    \item A confirmation message will appear and the new item will show in
      the list.
  \end{ugSteps}
\end{ugTask}

\begin{ugTask}{Edit an existing item}
  \begin{ugSteps}
    \item Navigate to the page containing the item you want to change.
    \item Click the three-dot button (\textbf{\ldots}) at the end of its row.
    \item Choose \textit{Edit} from the menu that appears.
    \item Make your changes in the pop-up window.
    \item Click the save button to confirm.
  \end{ugSteps}
\end{ugTask}

\begin{ugTask}{Delete an item}
  \begin{ugSteps}
    \item Navigate to the page containing the item you want to remove.
    \item Click the three-dot button (\textbf{\ldots}) at the end of its row.
    \item Choose \textit{Delete}.
    \item A confirmation message will ask you to confirm. Click to proceed.
  \end{ugSteps}
  \begin{ugWarning}
    Deleting an item often removes everything inside it. This cannot be
    undone.
  \end{ugWarning}
\end{ugTask}
