\subsection{Keyboard Shortcuts}

\begin{center}
\begin{tabular}{p{4cm}p{9cm}}
  \toprule
  \textbf{Key} & \textbf{What it does} \\
  \midrule
  \ugKey{Esc}              & Closes any open pop-up window without saving \\
  \ugKey{Tab}              & Moves the cursor to the next field in a form \\
  \ugKey{Enter}            & Submits the current form (in most pop-up windows) \\
  \ugKey{Shift} + \ugKey{Tab} & Moves the cursor back to the previous field \\
  \bottomrule
\end{tabular}
\end{center}
