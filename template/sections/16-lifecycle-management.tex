\section{Lifecycle Management}

\begin{ugModule}{Overview}

  \textbf{Purpose:}
  Manage the end-of-life cycle for documents, including soft
  deletion, restoration, and automatic archiving.

  \textbf{Who Can Access:}
  \ugRole{Admin} \ugRole{Manager}
\end{ugModule}

\subsection{Key Functions}
\begin{itemize}[leftmargin=18pt]
  \item \textbf{Soft Delete} --- Mark a document as inactive
    (\texttt{IsActive = false}). The document is hidden from
    active lists but remains in the database and storage.
  \item \textbf{Restore} --- Recover a soft-deleted document
    by setting \texttt{IsActive = true}.
  \item \textbf{Auto-Archive} --- Documents with an
    \ugField{Expiry Date} are automatically denied access
    after the date passes and can be transitioned to
    \ugStatus{ugAccent}{Archived} status.
  \item \textbf{Audit Trail} --- Every delete and restore
    action is recorded (who and when).
\end{itemize}

\subsection{Step-by-Step: Deleting a Document}
\begin{ugSteps}
  \item Navigate to the document detail page.
  \item Click the context menu (\,\faEllipsisV\,) and select
        \ugMenu{Delete}.
  \item Confirm the deletion in the modal dialog. The dialog
        shows the document title for verification.
  \item The document is moved to the Trash (soft-deleted).
\end{ugSteps}

\subsection{Step-by-Step: Restoring a Document}
\begin{ugSteps}
  \item Navigate to \ugMenu{Documents > Trash}.
  \item Locate the document using search or filters.
  \item Click \ugButton{Restore} on the document row.
  \item Confirm the restoration.
  \item The document reappears in the main document list with
        its original status, versions, and access controls
        intact.
\end{ugSteps}

\subsection{Automatic Expiry Behavior}

Documents with an \ugField{Expiry Date} follow this behavior:

\begin{tabularx}{\textwidth}{@{} l X @{}}
  \toprule
  \textbf{Event} & \textbf{System Behavior} \\
  \midrule
  Expiry date reached
    & Access via preview, download, and share links is
      automatically denied \\
  Expired document accessed
    & User sees an error: ``Document expired'' \\
  Admin action available
    & Extend the expiry date, archive the document,
      or upload a new version \\
  \bottomrule
\end{tabularx}

\subsection{Data Preservation}

\begin{ugWarning}
  Soft-deleted documents are \textbf{not} permanently removed.
  All file data, versions, access grants, audit logs, and
  download history are fully preserved. Permanent deletion
  is not available through the user interface --- this is by
  design to ensure compliance and data integrity.
\end{ugWarning}

\begin{ugTip}
  Periodically review the Trash to identify documents that
  may have been accidentally deleted. The Trash view supports
  the same search and filter capabilities as the main document
  list.
\end{ugTip}
