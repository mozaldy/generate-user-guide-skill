\subsection{Appendix B: Status Transition Rules}

\begin{tabularx}{\textwidth}{@{} l l l X @{}}
  \toprule
  \textbf{From} & \textbf{To} & \textbf{Who}
    & \textbf{Notes} \\
  \midrule
  Draft & Review
    & Owner
    & Submit for approval \\
  Review & Approved
    & \ugRole{Reviewer}
    & Approve the document \\
  Review & Rejected
    & \ugRole{Reviewer}
    & Reject with mandatory comment \\
  Approved & Active
    & \ugRole{Admin}, \ugRole{Manager}
    & Publish and make accessible \\
  Active & Archived
    & \ugRole{Admin}, \ugRole{Manager}
    & Manual archive or auto-expiry \\
  Rejected & Draft
    & Owner
    & Revise and resubmit \\
  \bottomrule
\end{tabularx}
